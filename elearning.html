# Buat fungsi untuk menghitung jumlah string yang panjangnya dua atau lebih dan memiliki karakter/huruf pertama dan terakhir yang sama
def count_strings(list_string):
    count = 0
    for string in list_string:
        if len(string) > 1 and string[0] == string[-1]:
            count += 1
    return count

# Panggil fungsi dan cetak hasil
print("Jumlah string yang panjangnya dua atau lebih dan memiliki karakter/huruf pertama dan terakhir yang sama:", count_strings(list_string))

# Buat fungsi untuk menghasilkan tiga nilai pertama dan tiga nilai terakhir
def get_first_last(list_x):
    first_three = list_x[:3]
    last_three = list_x[-3:]
    return first_three, last_three

# Panggil fungsi dan cetak hasil
first_three, last_three = get_first_last(list_x)
print("Tiga nilai pertama:", first_three)
print("Tiga nilai terakhir:", last_three)

# ini adalah soal nomor 5          
import math

def luas_persegi_panjang(panjang, lebar):
    return panjang * lebar

def luas_persegi(sisi):
    return sisi * sisi

def luas_segitiga(alas, tinggi):
    return 0.5 * alas * tinggi

def luas_lingkaran(jari_jari):
    return math.pi * jari_jari * jari_jari

def menu():
    print("============================================")
    print(" Menghitung Luas Bangun Datar ")
    print("============================================")
    print("1. Luas Persegi Panjang")
    print("2. Luas Persegi")
    print("3. Luas Segitiga")
    print("4. Luas Lingkaran")
    print("9. Exit")
    print("============================================")

def main():
    while True:
        menu()
        pilihan = int(input("Pilih Menu Menghitung Luas : "))
        
        if pilihan == 1:
            print("------ Menghitung Luas Persegi Panjang ------")
            panjang = float(input("Masukkan Panjang : "))
            lebar = float(input("Masukkan Lebar : "))
            luas = luas_persegi_panjang(panjang, lebar)
            print(f"Luas Persegi Panjang = {luas}")
        
        elif pilihan == 2:
            print("------ Menghitung Luas Persegi ------")
            sisi = float(input("Masukkan Nilai Sisi : "))
            luas = luas_persegi(sisi)
            print(f"Luas Persegi = {luas}")
        
        elif pilihan == 3:
            print("------ Menghitung Luas Segitiga ------")
            alas = float(input("Masukkan Alas : "))
            tinggi = float(input("Masukkan Tinggi : "))
            luas = luas_segitiga(alas, tinggi)
            print(f"Luas Segitiga = {luas}")
        
        elif pilihan == 4:
            print("------ Menghitung Luas Lingkaran ------")
            jari_jari = float(input("Masukkan Jari-jari : "))
            luas = luas_lingkaran(jari_jari)
            print(f"Luas Lingkaran = {luas}")
        
        elif pilihan == 9:
            print("Terima kasih telah menggunakan program ini.")
            break
        
        else:
            print("Pilihan tidak valid, silakan coba lagi.")

if __name__ == "__main__":
    main()
